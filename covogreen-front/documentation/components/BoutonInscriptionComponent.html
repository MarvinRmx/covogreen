import { Component, OnInit, Input  } from '@angular/core';
import {InscriptionTrajetService} from '../../services/inscription-trajet.service';
import {Response} from '@angular/http';
import {TrajetEnt} from "../../class/TrajetEnt";
import {JourneyService} from "../../services/journey.service";
import {UserService} from "../../services/user.service";
import {User} from '../../class/user';
import {isBoolean} from "util";

@Component({
  selector: 'app-bouton-inscription',
  templateUrl: './bouton-inscription.component.html',
  styleUrls: ['./bouton-inscription.component.css'],
  providers: [InscriptionTrajetService, UserService]
})
export class BoutonInscriptionComponent implements OnInit {
  	@Input() idTrajet: number;
  	inscrit: boolean;
  	connected: boolean;
  	messages: string[];
  	is_creator: boolean;

  	@Input() offre: TrajetEnt;

  	constructor(
  	    private inscriptionService: InscriptionTrajetService,
        private journeyService: JourneyService,
        private userService: UserService
    ) { }

	  ngOnInit() {
  		this.verifUserInscription();
  		this.isCreatorOfJourney();

      this.userService.getUser().subscribe((res: User) => {
        this.connected = (res.id_user !== 0)?true:false;
      });
  	}

  	/**
  	*	Envoi une requete vers le backend pour inscrire l'utilisateur connect au trajet X
  	*/
  	inscriptionTrajet(){
	  	// requete vers le back pour s'inscrire au trajet X
  		this.inscriptionService.inscription(this.idTrajet).subscribe((res: Response) => { // on rcupre la rponse.
        	// on vrifie la variable success
			if(res["success"] === true){
                this.inscrit = true;
                // On modifie le nombre de place.
                this.offre.nombre_place_disponible--;
			}else{
				this.messages = res["message"];
			}
    	});
  	}

  	/**
  	*	Envoi une requete vers le backend pour desinscrire l'utilisateur connect au trajet X
  	*/
  	desinscriptionTrajet(){
  		// requete vers le back pour se dsinscrire au trajet x
  		this.inscriptionService.desinscriptionTrajet(this.idTrajet).subscribe((res: Response) => { // on rcupre la rponse.// on vrifie la variable success
            if(res["success"] === true){
                this.inscrit = false;
                // On modifie le nombre de place.
                this.offre.nombre_place_disponible++;
            }else{
                this.messages = res["message"];
            }
    	});
	}


  	/**
  	*	Envoi une requete vers le backend pour dterminer si l'utilisateur X est inscrit au trajet Y.
  	*/
    verifUserInscription(){
        this.inscriptionService.verifInscription(this.idTrajet).subscribe( (res: Response) => {
        	this.inscrit = (res["success"] === false) ? true : false ;
        });
    }

    /**
     * @author Romain Lembo
     * Evite au crateur de l'offre de s'inscrire ou se dsinscrire du trajet
     */
    isCreatorOfJourney() {
        this.journeyService.isCreatorOfJourney(this.idTrajet)
            .subscribe( result => {
                console.log('isCreatorOfJourney : ', result);
                this.is_creator = result;
            });
    }
}

<div *ngIf="connected">
  <div *ngIf="!is_creator">

      <div class="inscription" *ngIf="!inscrit">
          <button class="btn btn-success btn-lg btn-block btnParticiper" (click)="inscriptionTrajet()" >
              <span>S'inscrire</span>
          </button>
      </div>

      <div class="desinscription" *ngIf="inscrit">
          <button class="btn btn-warning btn-lg btn-block btnParticiper" (click)="desinscriptionTrajet()" >
              <span>Se dsinscrire</span>
          </button>
      </div>

  </div>
</div>
